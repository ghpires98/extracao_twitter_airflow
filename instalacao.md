# Instalar a versão 3.9 do Python.

  sudo apt install python3.9

# Instalar o módulo de virtual enviroment da versão 3.9 do python.

  sudo apt install python3.9-venv

# Criar um enviroment. 

  python3.9 -m venv venv

# Ativar o enviroment. 

  source venv/bin/activate
